## Step 2: Copy and Paste This Code

Add this code **BEFORE** the closing `}` of the MainWindow class:

```csharp
        #region Menu Navigation Handlers

        /// <summary>
        /// Hides all content panels
        /// </summary>
        private void HideAllPanels()
        {
            WelcomePanel.Visibility = Visibility.Collapsed;
            BookingsPanel.Visibility = Visibility.Collapsed;
            VenuesPanel.Visibility = Visibility.Collapsed;
            RadioPanel.Visibility = Visibility.Collapsed;
            ChatPanel.Visibility = Visibility.Collapsed;
            ThemePanel.Visibility = Visibility.Collapsed;
            AdminPanel.Visibility = Visibility.Collapsed;
            SettingsPanel.Visibility = Visibility.Collapsed;
            HelpPanel.Visibility = Visibility.Collapsed;
        }

        /// <summary>
        /// Shows the specified panel and hides all others
        /// </summary>
        private void ShowPanel(Grid panel)
        {
            HideAllPanels();
            panel.Visibility = Visibility.Visible;
        }

        // Menu Click Handlers

        private void Menu_Bookings_Click(object sender, RoutedEventArgs e)
        {
            ShowPanel(BookingsPanel);
            // TODO: Load bookings data if needed
            // LoadBookingsData();
        }

        private void Menu_Venues_Click(object sender, RoutedEventArgs e)
        {
            ShowPanel(VenuesPanel);
            // TODO: Load venues data if needed
            // LoadVenuesData();
        }

        private void Menu_Radio_Click(object sender, RoutedEventArgs e)
        {
            ShowPanel(RadioPanel);
            // TODO: Initialize radio player if needed
        }

        private void Menu_Chat_Click(object sender, RoutedEventArgs e)
        {
            ShowPanel(ChatPanel);
            // Or open Chat window instead:
            // Chat_Click(sender, e);
        }

        private void Menu_Theme_Click(object sender, RoutedEventArgs e)
        {
            ShowPanel(ThemePanel);
        }

        private void Menu_Admin_Click(object sender, RoutedEventArgs e)
        {
            ShowPanel(AdminPanel);
        }

        private void Menu_Settings_Click(object sender, RoutedEventArgs e)
        {
            ShowPanel(SettingsPanel);
            // Or open settings window:
            // MySettings_Click(sender, e);
        }

        private void Menu_Help_Click(object sender, RoutedEventArgs e)
        {
            ShowPanel(HelpPanel);
        }

        #endregion
```
