## ?? IMPLEMENTATION EXAMPLE

### CandyBotWelcomeWindow.xaml.cs

```csharp
public partial class CandyBotWelcomeWindow : Window
{
    private CandyBotSoundManager _soundManager;
    
    public CandyBotWelcomeWindow()
    {
        InitializeComponent();
        _soundManager = new CandyBotSoundManager();
        
        // Play Desktop Mode intro on first launch
        if (IsFirstTimeLaunch())
        {
            PlayDesktopModeIntro();
        }
        else
        {
            PlayRandomWelcome();
        }
    }
    
    private void PlayDesktopModeIntro()
    {
        // Play one of the Desktop Mode feature lines
        var desktopModeLines = new[] { "062", "064", "067" };
        var random = new Random();
        var selectedLine = desktopModeLines[random.Next(desktopModeLines.Length)];
        _soundManager.PlayVoiceLine(selectedLine);
    }
    
    private void PlayRandomWelcome()
    {
        // Play any welcome message
        var welcomeLines = new[] { "061", "062", "063", "064", "065", "066", "067", "068", "069", "070" };
        var random = new Random();
        var selectedLine = welcomeLines[random.Next(welcomeLines.Length)];
        _soundManager.PlayVoiceLine(selectedLine);
    }
    
    private bool IsFirstTimeLaunch()
    {
        // Check if this is the first time launching
        var settings = Properties.Settings.Default;
        bool isFirst = !settings.HasLaunchedBefore;
        settings.HasLaunchedBefore = true;
        settings.Save();
        return isFirst;
    }
    
    private void DesktopModeButton_Click(object sender, RoutedEventArgs e)
    {
        // Play Desktop Mode highlight before opening
        _soundManager.PlayVoiceLine("064"); // "Try my Desktop Mode!"
        
        // Wait for voice line to finish
        System.Threading.Thread.Sleep(3000);
        
        // Open Desktop Widget
        var desktopWidget = new CandyBotDesktopWidget();
        desktopWidget.Show();
    }
}
```

```csharp
private void PlayWelcomeWithDesktopModePriority()
{
    var random = new Random();
    var priorityDesktopMode = IsWithinFirstWeek(); // Check launch date
    
    if (priorityDesktopMode)
    {
        // 60% chance for Desktop Mode lines
        if (random.Next(100) < 60)
        {
            var desktopModeLines = new[] { "062", "064", "067" };
            var line = desktopModeLines[random.Next(desktopModeLines.Length)];

        else

            _soundManager.PlayVoiceLine(line);

    {
        // Normal distribution
        PlayRandomWelcome();

private bool IsWithinFirstWeek()
{
    var installDate = Properties.Settings.Default.InstallDate;
    return (DateTime.Now - installDate).TotalDays <= 7;
}
```
