## STEP 3: ADD CONNECTION STATUS UPDATE CODE

In your `MainWindow.xaml.cs`, add these methods:

```csharp
#region Claude Wall of Honor - Connection Status Management

/// <summary>
/// Updates all Claude indicator connection statuses
/// </summary>
/// <param name="status">Status: "Connected", "Connecting", or "Disconnected"</param>
private void UpdateClaudeIndicators(string status)
{
    if (Claude001Indicator != null)
        Claude001Indicator.ConnectionStatus = status;
    
    if (Claude002Indicator != null)
        Claude002Indicator.ConnectionStatus = status;
}

/// <summary>
/// Call this when Cosmos DB connection succeeds
/// </summary>
private void OnDatabaseConnected()
{
    UpdateClaudeIndicators("Connected");
    ConnectionStatus.Text = "‚úÖ Connected to Azure Cosmos DB";
    ConnectionStatus.Foreground = new SolidColorBrush((Color)ColorConverter.ConvertFromString("#00ff41"));
}

/// <summary>
/// Call this when attempting to connect
/// </summary>
private void OnDatabaseConnecting()
{
    UpdateClaudeIndicators("Connecting");
    ConnectionStatus.Text = "üîÑ Connecting to Azure Cosmos DB...";
    ConnectionStatus.Foreground = new SolidColorBrush((Color)ColorConverter.ConvertFromString("#ffaa00"));
}

/// <summary>
/// Call this when connection fails
/// </summary>
private void OnDatabaseDisconnected()
{
    UpdateClaudeIndicators("Disconnected");
    ConnectionStatus.Text = "‚ùå Disconnected from Azure Cosmos DB";
    ConnectionStatus.Foreground = new SolidColorBrush((Color)ColorConverter.ConvertFromString("#ff4444"));
}

#endregion
```

```csharp
private async void Window_Loaded(object sender, RoutedEventArgs e)
{
    try
    {
        // Show connecting status
        OnDatabaseConnecting();
        
        // Your existing connection code

        // Show connected status
        OnDatabaseConnected();
        
        // Load your data
        await LoadVenuesAsync();
        await LoadBookingsAsync();
    }
    catch (Exception ex)
    {
        // Show disconnected status
        OnDatabaseDisconnected();
        
        MessageBox.Show($"Database connection failed: {ex.Message}", 
                       "Connection Error", 
                       MessageBoxButton.OK, 
                       MessageBoxImage.Error);
    }
}
```
