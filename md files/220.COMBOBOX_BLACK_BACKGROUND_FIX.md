## ?? **IMPLEMENTATION**

### **File Modified: `Themes/SpaceTheme.xaml`**

#### **Enhanced SpaceComboBox Style**
```xaml
<Style x:Key="SpaceComboBox" TargetType="ComboBox">
    <Setter Property="Background" Value="#000000"/>  <!-- Black -->
    <Setter Property="Foreground" Value="{DynamicResource AccentBrush}"/>  <!-- Green -->
    <Setter Property="BorderBrush" Value="{DynamicResource AccentBrush}"/>
    <Setter Property="BorderThickness" Value="2"/>
    <Setter Property="FontFamily" Value="Consolas"/>
    <Setter Property="FontSize" Value="14"/>
    <Setter Property="Padding" Value="10,5"/>
    
    <!-- Custom Template -->
    <Setter Property="Template">
        <Setter.Value>
            <ControlTemplate TargetType="ComboBox">
                <!-- Toggle Button (Arrow) -->
                <ToggleButton Background="#000000" 
                            BorderBrush="{DynamicResource AccentBrush}"/>
                
                <!-- Dropdown Popup -->
                <Popup x:Name="Popup">
                    <Border Background="#000000"  <!-- BLACK DROPDOWN -->
                          BorderBrush="{DynamicResource AccentBrush}"
                          BorderThickness="2">
                        <!-- Green glow effect -->
                        <Border.Effect>
                            <DropShadowEffect Color="{DynamicResource AccentColor}" 
                                            BlurRadius="15" 
                                            Opacity="0.8"/>
                        </Border.Effect>
                        
                        <!-- Items list -->
                        <ScrollViewer>
                            <StackPanel IsItemsHost="True" />
                        </ScrollViewer>
                    </Border>
                </Popup>
            </ControlTemplate>
        </Setter.Value>
    </Setter>
</Style>
```

#### **ComboBoxItem Style (Dropdown Items)**
```xaml
<Style x:Key="SpaceComboBoxItem" TargetType="ComboBoxItem">
    <Setter Property="Background" Value="#000000"/>  <!-- Black -->
    <Setter Property="Foreground" Value="{DynamicResource AccentBrush}"/>  <!-- Green -->
    <Setter Property="FontFamily" Value="Consolas"/>
    <Setter Property="Padding" Value="10,5"/>
    
    <Setter Property="Template">
        <Setter.Value>
            <ControlTemplate TargetType="ComboBoxItem">
                <Border Background="{TemplateBinding Background}">
                    <ContentPresenter />
                </Border>
                
                <ControlTemplate.Triggers>
                    <!-- Hover: Dark green -->
                    <Trigger Property="IsMouseOver" Value="True">
                        <Setter Property="Background" Value="#002200"/>
                    </Trigger>
                    
                    <!-- Selected: Darker green + bold -->
                    <Trigger Property="IsSelected" Value="True">
                        <Setter Property="Background" Value="#003300"/>
                        <Setter Property="FontWeight" Value="Bold"/>
                    </Trigger>
                </ControlTemplate.Triggers>
            </ControlTemplate>
        </Setter.Value>
    </Setter>
</Style>
```

#### **Default Application (All ComboBoxes)**
```xaml
<!-- Applied automatically to ALL ComboBoxes -->
<Style TargetType="ComboBox" BasedOn="{StaticResource SpaceComboBox}"/>
<Style TargetType="ComboBoxItem" BasedOn="{StaticResource SpaceComboBoxItem}"/>
```
