## ?? **CODE IMPLEMENTATION**

### Setting Personality with Age Check:

```csharp
public async Task SetPersonalityMode(CandyBotPersonalityMode mode)
{
    // Check if mode requires age verification
    if (mode == CandyBotPersonalityMode.Raunchy)
    {
        // Show age verification window
        var ageVerifyWindow = new AgeVerificationWindow();
        ageVerifyWindow.ShowDialog();
        
        if (!ageVerifyWindow.IsVerified)
        {
            MessageBox.Show(
                "Raunchy mode requires age verification (18+).\n\nMode change cancelled.",
                "Age Verification Required",
                MessageBoxButton.OK,
                MessageBoxImage.Warning);
            return;
        }
        
        // User is verified
        MessageBox.Show(
            "?? Raunchy Mode Activated! ??\n\nGet ready for a flirty experience!",
            "Personality Changed",
            MessageBoxButton.OK,
            MessageBoxImage.Information);
    }
    
    // Set the personality
    _candyBot.SetPersonality(mode);
    _preferences.PersonalityMode = mode;
    
    // Save preference
    await SavePreferences();
    
    // Notify user
    string modeName = mode switch
    {
        CandyBotPersonalityMode.Normal => "?? Normal",
        CandyBotPersonalityMode.Shy => "?? Shy",
        CandyBotPersonalityMode.Funny => "?? Funny",
        CandyBotPersonalityMode.ShitStirring => "?? Shit-Stirring",
        CandyBotPersonalityMode.Raunchy => "?? Raunchy",
        CandyBotPersonalityMode.Professional => "?? Professional",
        _ => "Normal"
    };
    
    // Show confirmation (if not Raunchy - already shown)
    if (mode != CandyBotPersonalityMode.Raunchy)
    {
        MessageBox.Show(
            $"Personality changed to: {modeName}",
            "Mode Changed",
            MessageBoxButton.OK,
            MessageBoxImage.Information);
    }
}
```

### Getting Time-Based Greetings:

```csharp
public string GetGreeting(string userName)
{
    var hour = DateTime.Now.Hour;
    string timeOfDay;
    
    if (hour >= 5 && hour < 12)
        timeOfDay = "morning";
    else if (hour >= 12 && hour < 17)
        timeOfDay = "afternoon";
    else if (hour >= 17 && hour < 22)
        timeOfDay = "evening";
    else
        timeOfDay = "latenight";
    
    // Get personality-specific greeting
    return GetPersonalityGreeting(_currentPersonality, timeOfDay, userName);
}
```
