### Step 4: Deletion Process
```csharp
// Delete user account
await _cosmosDbService.DeleteUserAsync(_currentUser.Id);

// Delete all user's bookings
var bookings = await _cosmosDbService.GetAllBookingsAsync();
foreach (var booking in bookings)
{
    if (booking.DJName == _currentUser.Username)
    {
        await _cosmosDbService.DeleteBookingAsync(booking.Id);
    }
}

// Logout
LogoutAndClose();
```
