```csharp
var auth = AuthenticationService.Instance;

    // Not logged in
}
```

### **Get Current User**
```csharp
var currentUser = auth.CurrentUser;
string username = currentUser?.Username;
```

### **Check Permission**

## ?? **DATABASE OPERATIONS**

### **Get Users**
```csharp
var users = await _cosmosService.GetAllUsersAsync();
```

### **Get User by Username**
```csharp
var user = await _cosmosService.GetUserByUsernameAsync("username");
```

### **Add User**
```csharp
await _cosmosService.AddUserAsync(newUser);
```

### **Update User**
```csharp
await _cosmosService.UpdateUserAsync(user);
```

### **Delete User**
```csharp
await _cosmosService.DeleteUserAccountAsync("username");
```
