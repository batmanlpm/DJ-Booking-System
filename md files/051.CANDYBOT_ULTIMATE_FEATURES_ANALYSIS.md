                Type = "celebration",
                Message = "?? One more booking and you'll hit 50 total bookings! You're crushing it!",

public class ProactiveAssistant
{
    // Checks happening in background
    public async Task MonitorAndNotify(User user)
    {
        // Before a booking
        var upcomingBookings = await GetUpcomingBookings(user.Username);
        foreach (var booking in upcomingBookings)
        {
            var daysUntil = (booking.Date - DateTime.Now).Days;

            if (daysUntil == 7)
                await Notify("?? Your set at {booking.Venue} is in ONE WEEK! Time to prepare your playlist! ??");

            if (daysUntil == 1)
                await Notify("? Tomorrow's the big day! Your set at {booking.Venue} is at {booking.Time}. Break a leg! ??");

            if (daysUntil == 0 && (booking.Time - DateTime.Now).Hours == 2)
                await Notify("?? 2 HOURS until showtime at {booking.Venue}! Test your equipment and get hyped! ??");
        }

        // After a booking
        var recentBookings = await GetBookingsInLastWeek(user.Username);
        foreach (var booking in recentBookings.Where(b => !b.FeedbackProvided))
        {
            if ((DateTime.Now - booking.Date).Days == 1)
                await AskForFeedback(booking);
        }

        // Check for conflicts
        var conflicts = await DetectScheduleConflicts(user.Username);
        if (conflicts.Any())
            await AlertConflict(conflicts.First());

        // Check venue requirements
        var nextBooking = upcomingBookings.OrderBy(b => b.Date).FirstOrDefault();
        if (nextBooking != null)
        {
            var venue = await GetVenue(nextBooking.VenueId);
            if (venue.RequiresEquipment && !user.EquipmentConfirmed)
                await Notify($"?? Hey! {venue.Name} requires you to bring {venue.EquipmentList}. Do you have everything?");
        }
    }

    public async Task GivePersonalizedTips(User user)
    {
        var stats = await AnalyzePerformanceStats(user.Username);

        if (stats.AverageRating < 4.0)
            await SendTip("?? Pro Tip: Start with crowd-pleasers! Check our top 10 most-played tracks.");

        if (stats.BookingsThisMonth == 0)

        if (stats.PrimeTimeBookingPercentage < 20)
            await SendTip("? Did you know prime-time slots (9PM-1AM) get 3x more audience? Try booking one!");

}
